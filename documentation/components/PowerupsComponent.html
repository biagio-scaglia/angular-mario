import { Component } from '@angular/core';

import { MyButtonComponent } from '../shared/mario-button/mario-button.component';

@Component({
  selector: 'app-powerups',
  standalone: true,
  imports: [RouterLink, MyButtonComponent],
  templateUrl: './powerups.component.html',
  styleUrls: ['./powerups.component.css']
})
export class PowerupsComponent {
  powerups = [

      id: 'fungo',
      name: 'Fungo ',

        <h2>{{ powerup.name }}</h2>
        <div class="powerup-type">{{ powerup.type }}</div>
        <p class="powerup-short-description">{{ powerup.shortDescription }}</p>
        <app-mario-button [routerLink]="['/powerups', powerup.id]" [cta]="'Scopri di più'" class="powerup-button">
