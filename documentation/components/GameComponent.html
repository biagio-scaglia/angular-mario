import { Component } from '@angular/core';

@Component({
  selector: 'app-game',
  standalone: true,
  imports: [MyButtonComponent, RouterLink],
  templateUrl: './game.component.html',
  styleUrls: ['./game.component.css']
})
export class GameComponent {
  games = [
    {
      title: "Mario Odyssey",
      coverImage: "mario-odyssey.jpg",

}
